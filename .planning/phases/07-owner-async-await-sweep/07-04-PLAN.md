    if ((typeof (owner) === "undefined") || (owner === null)) {
      return callback(false, "undefined_owner");
    }

      this.saveAvatar(owner, body.avatar, (success, err) => {
        callback(success, err);
      });
      return;
    }
    this.process_update(body, (update_key, update_value) => {
      alog.log(owner, "Attempt to update owner: " + owner + " with: " + update_key);

        console.log("🔨 [debug] invalid_protocol_update_key_missing in", body);

      this.apply_update(owner, update_key, update_value, callback);
    });
  }

Async/await target shape:
  async update(owner, body, callback) {
    if ((typeof (owner) === "undefined") || (owner === null)) {
      return callback(false, "undefined_owner");
    }
    if (typeof (body.avatar) !== "undefined") {
      // saveAvatar is OUT-OF-Phase-7 scope — stays callback-style
      this.saveAvatar(owner, body.avatar, (success, err) => {
        callback(success, err);
      });
      return;
    }
    // process_update is pure-sync, callback-style (Plan 07-01 left it non-async)
    this.process_update(body, (update_key, update_value) => {
      alog.log(owner, "Attempt to update owner: " + owner + " with: " + update_key);
      if ((typeof (update_key) === "undefined") || (update_key === null)) {
        console.log("🔨 [debug] invalid_protocol_update_key_missing in", body);
        return callback(false, "invalid_protocol_update_key_missing");
      }
      // apply_update is async (Plan 07-01) but its public signature stays
      // apply_update(owner, update_key, update_value, callback) — direct callback-passthrough works.
      this.apply_update(owner, update_key, update_value, callback);
    });
  }

Observation: Owner.update has NO `this.userlib.*` calls of its own — all DB work is delegated to apply_update. The conversion is therefore minimal: add `async ` to the outer signature and keep the body essentially unchanged. The value of the conversion is consistency (the entire file is now async) AND the behavior-locking unit test. The async keyword on update is technically not REQUIRED (the method does no awaiting itself), BUT adding it matches the file-wide pattern and signals intent. Decision: ADD `async` for consistency.
