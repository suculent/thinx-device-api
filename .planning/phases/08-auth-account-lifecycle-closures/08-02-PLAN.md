  // ... fetch user doc by owner via this.userlib.view("users", "owners_by_username") ...
  // ... handle "user_not_found" / "missing_reset_key" / etc. ...
  let user = body.rows[0].doc;
  let user_reset_key = user.reset_key;
  if (typeof (user_reset_key) === "undefined") {
    user_reset_key = null;
  }
  console.log(`ℹ️ [info] Attempting to reset password with key ${Util.redactToken(reset_key)}`);
  if (reset_key !== user_reset_key) {
    console.log("☣️ [error] reset_key does not match");
    callback(false, "invalid_reset_key");

  }
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Email recipient → API GET /api/user/password/reset | Untrusted HTTP client clicks a link from email; `reset_key` and `owner` come in via query string. |
| API GET handler → Vue console redirect target | The API returns a 302 with `Location: <Vue console URL>`. The Location header value is constructed from `app_config.public_url` (trusted config) + fixed path + `reset_key` + `owner` (query-string echoed). |
| Vue console `/password-reset` page → POST /password/reset | Vue submits `{ reset_key, password, rpassword }` to the existing POST endpoint, which runs the actual password-change logic. Outside this phase's scope. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08.2-01 | Tampering | Redirect URL construction at owner.js:506 | mitigate | The URL is composed from `app_config.public_url` (server-controlled trusted config), a hardcoded path literal `/password-reset?reset_key=`, plus `reset_key` and `owner` values that have ALREADY been validated by the success-branch precondition (`reset_key === user_reset_key` and `owner` matched a user doc in CouchDB). No user-controlled string is injected into the host or path portion of the URL. |
| T-08.2-02 | Open Redirect | API GET handler 302 Location | mitigate | The Location header points only at `app_config.public_url` (server-controlled) — never at a user-supplied URL. There is NO open-redirect surface here; `reset_key` and `owner` flow into query parameters of a server-fixed host+path, not into the host itself. |
| T-08.2-03 | Information Disclosure | reset_key in Location header / browser history | accept | The Location header echoes `reset_key` (unchanged from today's behavior — pre-existing condition since `/password.html?reset_key=...` already exposed this). The `Util.redactToken` discipline (SEC-PII-01) covers SERVER-SIDE log lines only; the URL itself necessarily carries the key as a one-time-use token that the Vue page consumes and discards. The reset_key is single-use and short-lived; the existing flow's exposure profile is unchanged. |
| T-08.2-04 | Tampering | The strict-equality reset_key check (Phase 5 REFACTOR-02) | mitigate | The `if (reset_key !== user_reset_key)` branch at owner.js:502 is preserved verbatim by Task 1's explicit "do not modify the strict-equality branch" instruction. Must_have #2 binds this. |
| T-08.2-05 | Repudiation | Password-reset chain accountability | accept | No new audit log entry added in this phase — the existing audit trail (reset_key generation in `password_reset_init`, set_password completion in `set_password_reset`) is unchanged. Sufficient for the redirect-only change. |
| T-08.2-06 | Spoofing | Email-link phishing (attacker sends fake "reset link" to a victim) | accept | The redirect target is server-controlled; an attacker cannot redirect victims to an attacker-controlled URL via this change. | T-08.2-07 | Denial of Service | Vue console `/password-reset` route availability | accept | Pre-existing route; this phase's redirect change does not increase load characteristics. If the Vue console is down, the redirect still resolves to a broken page — same failure mode as today's `/password.html` if it were unserved. |
| T-08-SC | Tampering | npm/pip/cargo installs | accept | No new package installs in this phase. RESEARCH.md Package Legitimacy Audit therefore N/A. |
